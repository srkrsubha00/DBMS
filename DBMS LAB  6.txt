Create table Customers(
CustomerID Number (1) PRIMARY KEY,
CustomerName Varchar2 (20),
ContactName Varchar2 (20),
Address Varchar2 (25),
City Varchar2 (10),
PostalCode Number (5),
Country Varchar2 (10));

Create table Suppliers(
SupplierID Number (1) PRIMARY KEY,
SupplierName Varchar2 (30),
ContactName Varchar2 (20),
Address Varchar2 (25),
City Varchar2 (10),
PostalCode Number (5),
Country Varchar2 (10));
